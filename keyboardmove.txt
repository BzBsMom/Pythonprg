# example of using the keyboard to move around.import simpleguiWidth=600Height=400Ball_radius=20Ball_pos= [Width/2, Height/2]# Handlersdef keydown(key):    vel=5            if key == simplegui.KEY_MAP['left']:        Ball_pos[0] -= vel    if key == simplegui.KEY_MAP['right']:        Ball_pos[0] += vel    Ball_pos[0] = Ball_pos[0] % Width    if key == simplegui.KEY_MAP['down']:        Ball_pos[1] += vel    if key == simplegui.KEY_MAP['up']:        Ball_pos[1] -= vel    Ball_pos[1] = Ball_pos[1] % Height        def draw(canvas):    canvas.draw_circle(Ball_pos,Ball_radius, 4, "Red", "White")# Create a frame and assign callbacks to event handlersframe = simplegui.create_frame("Move", Width, Height)#register event handlersframe.set_keydown_handler(keydown)frame.set_draw_handler(draw)# Start the frame animationframe.start()
