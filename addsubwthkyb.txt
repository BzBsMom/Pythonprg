# example of using the keyboard.import simpleguicurrent_val=5times_press=0# Handlersdef keydown(key):    global current_val,times_press    times_press+=1    current_val+=current_val        print "times press",times_press    print current_valdef keyup(key):    global current_val    current_val-=3        print current_val# Create a frame and assign callbacks to event handlersframe = simplegui.create_frame("add & sub", 35, 35)#register event handlersframe.set_keydown_handler(keydown)frame.set_keyup_handler(keyup)# Start the frame animationframe.start()
