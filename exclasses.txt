FEES = 5class BankAccount:    def __init__(self, initial_balance):        self.balance = initial_balance        self.fees = 0          def deposit(self, amount):        self.balance += amount           def withdraw(self, amount):         if self.balance < amount:             self.fees += FEES             self.balance -= FEES         self.balance -= amount            def get_balance(self):        return self.balance        def get_fees(self):        return self.fees    account1 = BankAccount(10)account1.withdraw(15)account2 = BankAccount(15)account2.deposit(10)account1.deposit(20)account2.withdraw(20)print account1.get_balance(), account1.get_fees(), account2.get_balance(), account2.get_fees()
