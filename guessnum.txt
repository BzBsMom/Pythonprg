Range from", low,"to",high    print "Number of remaining guesses is", num_tries# define event handlers for control paneldef range100():    # button that changes range to range [0,100) and restarts    global low, high, num_tries    low=0    high=100    num_tries=7    new_game()    def range1000():    # button that changes range to range [0,1000) and restarts    global low, high, num_tries    low=0    high=1000    num_tries=10    new_game()        def input_guess(guess):    # main game logic goes here	    global num_tries    guess_num = int(guess)        print ""    print "Guess was",guess    num_tries -=1    print "Number of remaining guesses is", num_tries    if guess_num > computer_guess and num_tries > 0:       print "Lower !"    elif guess_num < computer_guess and num_tries > 0:       print "Higher !"    elif guess_num == computer_guess and num_tries > 0:       print "Correct !"       new_game()    else:       print "You ran out of guesses. The number was",computer_guess       new_game()     # create frameframe = simplegui.create_frame('What is my number?', 200, 200)frame.set_canvas_background('Blue')# register event handlers for control elementsbutton1 = frame.add_button('range100', range100)button2 = frame.add_button('range1000', range1000)inp = frame.add_input('Guess', input_guess, 50)# call new_game and start framenew_game()frame.start()# always remember to check your completed program against the grading rubric
